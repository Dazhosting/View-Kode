import fs from "node:fs";
import fetch from "node-fetch";
import FormData from "form-data";

async function removeBg(imageURL) {
  const formData = new FormData();
  formData.append("size", "auto");
  formData.append("image_url", imageURL);

  const response = await fetch("https://api.remove.bg/v1.0/removebg", {
    method: "POST",
    headers: {
      "X-Api-Key": "BbYH9jSRgrQcMG1ErEeoRkLK"
    },
    body: formData,
  });

  if (response.ok) {
    const buffer = await response.arrayBuffer();
    return Buffer.from(buffer);
  } else {
    const errorText = await response.text();
    throw new Error(`Remove.bg Error ${response.status}: ${response.statusText}\n${errorText}`);
  }
}

// Contoh Pemakaian Nya ini
(async () => {
  try {
    const imageURL = "https://example.jpg"; // ganti dengan URL gambar Kalian nyaaa
    const resultBuffer = await removeBg(imageURL);
    fs.writeFileSync("no-bg.png", resultBuffer);
    console.log("✅ Background berhasil dihapus dan disimpan sebagai 'no-bg.png'");
  } catch (error) {
    console.error("❌ Gagal menghapus background:", error.message);
  }
})();
